#PWM using RPi.GPIO library (I didnt know this!!!)
import RPi.GPIO as GPIO
import time

pwm_pin = 18 #Broadcom (BCM) pin 18 (pin 16 on Pi)
dc = 100 #Duty Cycle of PWM (0 - 100)
GPIO.setwarnings(False)
GPIO.setmode(GPIO.BCM)
GPIO.setup(pwm_pin, GPIO.OUT)
pwm = GPIO.PWM(pwm_pin, 50) #Sets PWM frequency to 50Hz
print('pwm is running')
while 1:
        pwm.start(dc)
